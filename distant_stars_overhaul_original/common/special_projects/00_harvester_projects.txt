special_project = {
	key = "HARVESTER_DISTURBANCE_1_PROJECT"
	event_chain = "harvest_subterranean_civilization_chain"
	cost = 1
	tech_department = society_technology
	picture = GFX_evt_underground_civilization
	timelimit = 360
	
	event_scope = planet_event
	
	on_success = {
		root = { change_pc = pc_bh_harvest_colony }
		space_owner = { country_event = { id = harvest_colony.510 } }
		space_owner = { end_event_chain = "harvest_subterranean_civilization_chain" }
	}
	
	on_fail = {
		end_event_chain = "harvest_subterranean_civilization_chain"
		random_list = {
			60 = {
				root = { planet_event = { id = harvest_colony.54 } }
			}
			40 = {
				root = { planet_event = { id = harvest_colony.59 } }
			}
		}
	}
}

special_project = {
	key = "HARVESTER_DISTURBANCE_2_PROJECT"
	event_chain = "harvest_subterranean_civilization_chain"
	cost = 1
	tech_department = engineering_technology
	picture = GFX_evt_nuclear_explosion
	timelimit = 30
	
	same_option_group_as = { HARVESTER_DISTURBANCE_1_PROJECT }
	
	event_scope = planet_event
	
	on_success = {
		planet_event = { id = harvest_colony.52 }
		space_owner = { end_event_chain = "harvest_subterranean_civilization_chain" }
	}
	
	on_fail = {
		end_event_chain = "harvest_subterranean_civilization_chain"
		if = {
			limit = {
				root = {
					space_owner = {
						is_homicidal = yes
					}
				}
			}
			root = { planet_event = { id = harvest_colony.54 } }
		}
	}
}
