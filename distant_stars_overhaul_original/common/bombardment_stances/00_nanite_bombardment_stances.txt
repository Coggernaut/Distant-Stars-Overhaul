nanite_sterilization = {
	trigger = {
		owner = {
			has_technology = tech_nanite_torpedo
		}
	}

	# icon_frame = 3
	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 0
	army_damage = 2.0

	kill_pop_chance = { base = 1.0 }
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
		modifier = {
			factor = 20
			exists = owner
			owner = { has_civic = civic_fanatic_purifiers }
		}
		modifier = {
			factor = 20
			exists = from.owner
			exists = owner
			from.owner = { is_machine_empire = no }
			owner = { has_civic = civic_machine_terminator }
		}
	}
}
