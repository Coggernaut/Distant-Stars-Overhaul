building_nanite_assembly_plant = {
	base_buildtime = 360
	icon = building_robot_assembly_plant
	can_build = no

	category = pop_assembly

	prerequisites = {
		tech_mega_assembly
		"tech_nanites_assembler_factory"
	}

	potential = {
		exists = owner
		owner = { is_machine_empire = yes }
	}

	convert_to = { building_machine_assembly_complex }

	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				num_buildings = { type = any value <= 14 }
				merg_is_habitat = no
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			# buildings_no_crystals = yes
			owner = { is_robot_empire = no }
			AND = {
				owner = { is_ai = yes }
				free_jobs < 100
				OR = {
					num_buildings = { type = any value = 16 }
					merg_is_habitat = yes
				}
			}
		}
	}

	planet_modifier = { job_replicator_add = 500 }

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
			nanites = 50
		}
		upkeep = {
			energy = 12
			nanites = 1
		}
	}

	triggered_desc = { text = job_replicator_effect_desc }

	ai_weight = {
		weight = 50000
		modifier = {
			factor = 0
			OR = {
				merg_is_habitat = yes
				num_buildings = { type = any value >= 14 }
			}
		}
	}
}
